681  bool validate_spatial_axis (Math::SpatialVector &axis)
682  {
683 #ifdef RBDL_USE_CASADI_MATH
684  // If using casadi, the axes won't be validated
685  return true;
686 #else
687  bool axis_rotational = false;
688  bool axis_translational = false;
689 
690  Math::Vector3d rotation (axis[0], axis[1], axis[2]);
691  Math::Vector3d translation (axis[3], axis[4], axis[5]);
692 
693  if (fabs(rotation.norm()) > 1.0e-8) {
694  axis_rotational = true;
695  }
696 
697  if (fabs(translation.norm()) > 1.0e-8) {
698  axis_translational = true;
699  }
700 
701  if(axis_rotational && rotation.norm() - 1.0 > 1.0e-8) {
702  std::cerr << "Warning: joint rotation axis is not unit!" << std::endl;
703  }
704 
705  if(axis_translational && translation.norm() - 1.0 > 1.0e-8) {
706  std::cerr << "Warning: joint translation axis is not unit!" << std::endl;
707  }
708 
709  return axis_rotational != axis_translational;
710 #endif
711  }
